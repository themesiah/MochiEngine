101 inline Rectf AnchoredPosition(const std::optional<Rectf> &parentRect, const Rectf &dstRect, const Vector2f &parentAnchor, const Vector2f &spritePivot)
102 {
103 Rectf parent;
104 if (parentRect)
105 {
106 parent = parentRect.value();
107 parent.SetPosition(parent.GetPosition() + parent.GetSize() / 2.0f);
108 }
109 else
110 {
111 Vector2f logical{static_cast<float>(CONST_RENDER_LOGICAL_X), static_cast<float>(CONST_RENDER_LOGICAL_Y)};
112 parent = {logical / 2.0f, logical};
113 }
114 auto elementSize = dstRect.GetSize();
115 Vector2f pos = (parent.GetSize() * parentAnchor) / 2.0f + parent.GetPosition();
116 pos -= (elementSize / 2.0f) * spritePivot + (elementSize / 2.0f);
117 pos += dstRect.GetPosition();
118
119 return {pos, dstRect.GetSize()};
120 }
